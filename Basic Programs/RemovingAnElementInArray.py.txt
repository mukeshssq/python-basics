#include<stdio.h>
int main()
{
    int n,i,a[50];
    int index;
    printf("enter the array size; ");
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("enter the element to delete:");
    scanf("%d",&index);
    if(index>n)
    {
     printf("element is not found");
     return 0;
    }
    for(i=index;i<n;i++)
    {
        a[i]=a[i+1];
        
    }
    printf("ouput:\n");
    for(i=0;i<n-1;i++)
    {
        printf("->%d\n", a[i]);
        
    }
    return 0;
    
}
